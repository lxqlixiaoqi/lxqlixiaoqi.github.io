4. 使用方法

// 1. 引入工具类
<script src="scripts/json-utils.js"></script>

// 2. 安全解析JSON
const jsonString = '{"name": "蘑菇", "age": 18}';
const result = JsonUtils.safeParse(jsonString, {
    tryFix: true,  // 尝试修复格式错误
    defaultValue: {},  // 解析失败时的默认值
    onError: function(error) {
        console.error('解析JSON失败:', error);
    }
});

// 3. 处理服务器响应
const serverResponse = '{"status": "success", "data": {...}}';
const processedData = JsonUtils.processResponse(serverResponse);
